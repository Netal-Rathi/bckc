select  p.product_id  ,
Round(case when count(p.product_id)=null then 0.00 else
 SUM(p.price*u.units)/sum(u.units) end,2 )as average_price
from Prices p
LEFT join UnitsSold u on
p.product_id=u.product_id
AND  u.purchase_date between p.start_date AND p.end_date
GROUP BY p.product_id  ;
