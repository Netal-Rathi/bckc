select  p.product_id  ,
Round(
    COALESCE(
    case when sum(u.units)=0 then 0.00 else
 SUM(p.price*u.units)/sum(u.units) end,0),2
  )as average_price
from Prices p
LEFT join UnitsSold u on
p.product_id=u.product_id
AND  u.purchase_date between p.start_date AND p.end_date
GROUP BY p.product_id  ;
