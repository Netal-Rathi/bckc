WITH first_login AS (
    SELECT 
        player_id, 
        MIN(event_date) AS first_date
    FROM Activity
    GROUP BY player_id
),
logged_in_next_day AS (
    SELECT 
        f.player_id
    FROM first_login f
    JOIN Activity a
    ON f.player_id = a.player_id
       AND a.event_date = DATE_ADD(f.first_date, INTERVAL 1 DAY)
)
SELECT 
    ROUND(COUNT(DISTINCT l.player_id) * 1.0 / COUNT(DISTINCT a.player_id), 2) AS fraction
FROM logged_in_next_day l
CROSS JOIN Activity a;
