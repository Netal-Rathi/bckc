with first_logged_in as (
    select player_id ,
     min(event_date) as first_date
     from Activity 
     group by player_id

),
 consecutive_dates as (
    select f.player_id 
    from first_logged_in f 
    join Activity a 
    on a.player_id=f.player_id
   AND a.event_date = DATE_ADD(f.first_date, INTERVAL 1 DAY)

)
select 
Round(
   IFNULL(count(distinct l.player_id)/count(distinct a.player_id),0),2) as fraction
    from consecutive_dates l cross join Activity a ;
