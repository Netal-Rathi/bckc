SELECT 
    ROUND(
        (SUM(CASE WHEN d.order_date = d.customer_pref_delivery_date THEN 1 ELSE 0 END) / COUNT(d.customer_id)) * 100, 
        2
    ) AS immediate_percentage
FROM Delivery d
WHERE d.order_date IN (
    SELECT MIN(order_date) 
    FROM Delivery
    GROUP BY customer_id
);
