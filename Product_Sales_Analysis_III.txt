with first_years as(
    select min(year) as first_year from
    Sales
    group by year
)
select s.product_id , f.first_year as first_year ,s.quantity,s.price
from Sales s join Product p 
on s.product_id=p.product_id
join first_years f on f.first_year=s.year;
