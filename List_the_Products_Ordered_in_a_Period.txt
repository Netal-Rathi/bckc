SELECT 
    p.product_name,
    SUM(o.unit) AS unit
FROM 
    Products p
JOIN 
    Orders o 
    ON p.product_id = o.product_id
WHERE 
    STR_TO_DATE(o.order_date, '%d-%m-%Y') BETWEEN '2020-02-01' AND '2020-02-29'
GROUP BY 
    p.product_name
HAVING 
    SUM(o.unit) >= 100;

// Title: List the Products Ordered in a Period
