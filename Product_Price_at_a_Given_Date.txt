WITH t1 AS (
    SELECT product_id, MAX(new_price) AS price
    FROM products
    WHERE change_date <= '2019-08-16'
    GROUP BY product_id
),
t2 AS (
    SELECT product_id, 10 AS price
    FROM products
    WHERE change_date > '2019-08-16'
)
SELECT product_id, MAX(price) AS price
FROM (
    SELECT * FROM t1
    UNION ALL
    SELECT * FROM t2
) combined
GROUP BY product_id;
